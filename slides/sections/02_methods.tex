\section{Methods and Algorithm Mechanics}

\begin{frame}{Algorithm Family Overview}
\small
\begin{tabular}{>{\bfseries}p{0.22\linewidth}p{0.73\linewidth}}
Naive & Basic 2D sliding-window exact matching baseline. \\
Rabin-Karp & 2D rolling hash over windows, then exact verification \cite{karp1987rabin}. \\
Vectorized RK & Same RK logic with NumPy-wide window hashing. \\
Greedy KMP & Match pattern first row with KMP, then naively verify remaining pattern rows \cite{knuth1977fast}. \\
Greedy BM & Match pattern first row with BM, then naively verify remaining pattern rows \cite{boyer1977fast}. \\
Bird-KMP & Column-encoding + Aho-Corasick-like column scan + KMP over IDs \cite{bird1977twod,aho1975efficient}. \\
Bird-BM & Same Bird preprocessing, BM over column-ID sequence. \\
\end{tabular}
\end{frame}

\begin{frame}{What We Measured}
\begin{itemize}
  \item Every algorithm returns \texttt{(comparison\_count, match\_position)}.
  \item \texttt{Counter} wraps equality checks; this is the primary metric used in analysis.
  \item Wall-clock is captured externally per run as secondary runtime evidence.
\end{itemize}
\begin{block}{Interpretation rule}
\small
\textbf{comparison\_count} is primary (algorithm behavior), while \textbf{wall-clock} is secondary (implementation runtime evidence).
\end{block}
\end{frame}

\begin{frame}{Naive Search: Core Mechanics}
\small
\textbf{Input:} text \(T\in\{0,1\}^{N\times N}\), pattern \(P\in\{0,1\}^{M\times M}\)
\begin{enumerate}
  \item Enumerate every valid top-left window \((i,j)\).
  \item Compare all \(M^2\) pattern cells against \(T[i:i+M, j:j+M]\).
  \item Stop at first full match, else return \texttt{None}.
\end{enumerate}
\end{frame}

\begin{frame}{Behavior of Naive Search}
\small
\begin{itemize}
  \item Observed Strength: very transparent baseline and stable implementation.
  \item Observed Weakness: no pruning; cost grows quickly when no early match occurs.
  \item Implementation details: minimal overhead makes it a useful control for interpreting other methods.
\end{itemize}
\end{frame}

\begin{frame}{2D Rabin-Karp: Rolling Hash Pipeline}
\small
\begin{enumerate}
  \item Compute pattern hash using row and column polynomial bases.
  \item Build row-wise rolling hashes for text.
  \item Build vertical rolling hashes over those row hashes.
  \item On hash hit, perform exact cell verification.
\end{enumerate}
\begin{block}{Why it helps}
Most windows are rejected cheaply by hash mismatch before full comparison.
\end{block}
\end{frame}

\begin{frame}{Vectorized RK}
\small
\begin{itemize}
  \item Uses NumPy sliding windows and reduction operations to hash many windows together.
  \item Reduces Python-loop overhead in hash computation stage.
\end{itemize}
\begin{block}{Expected effect}
Lower runtime than loop RK when vectorized operations amortize setup overhead.
\end{block}
\end{frame}

\begin{frame}{Greedy KMP}
\small
\begin{enumerate}
  \item Build LPS array from first pattern row.
  \item Run KMP to match the first pattern row against each candidate text row.
  \item For every first-row hit, naively verify the remaining pattern rows at the same column.
\end{enumerate}
\begin{itemize}
  \item Effective when first-row mismatches are frequent.
  \item Can overwork when first row matches often but full pattern fails.
\end{itemize}
\end{frame}

\begin{frame}{Greedy BM}
\small
\begin{enumerate}
  \item Build bad-character table from first pattern row.
  \item Match right-to-left on that row; shift aggressively on mismatch.
  \item If the first row matches, naively verify the remaining pattern rows.
\end{enumerate}
\begin{block}{Observed Implementation Effect}
Greedy BM benefits when first-row mismatches are frequent (random-like data), but degrades when many top rows trigger verification or close matches are frequent.
\end{block}
\end{frame}

\begin{frame}{Bird Methods}
\small
\begin{enumerate}
  \item Encode pattern columns into IDs.
  \item Build automaton over unique column signatures.
  \item For each top-row offset, derive a column-ID sequence of text windows.
  \item Run 1D matcher over IDs: KMP variant or BM variant.
\end{enumerate}
\begin{block}{Goal}
Reduce expensive full-grid matching to structured column-sequence matching.
\end{block}
\end{frame}

\begin{frame}{Bird-KMP and Bird-BM: Tradeoffs}
\small
\begin{itemize}
  \item Usually excellent in comparison-count efficiency (especially Bird-BM).
  \item Preprocessing and automaton traversal add heavy constant-time overhead.
  \item This overhead becomes non-negligible as text/pattern sizes increase.
  \item In our profiling, \texttt{\_build\_C\_row} and \texttt{\_ac\_next} dominate runtime.
\end{itemize}
\end{frame}

\begin{frame}{Why Comparison and Runtime Rankings Diverge}
\small
\begin{tabular}{>{\bfseries}p{0.27\linewidth}p{0.66\linewidth}}
Counted work & Equality checks routed through \texttt{Counter}. \\
Non-counted work & Hash updates, dict lookups, automaton transitions, NumPy reductions, memory traversal. \\
Outcome & Algorithms with fewer counted comparisons can still be slower overall. \\
\end{tabular}
\end{frame}
