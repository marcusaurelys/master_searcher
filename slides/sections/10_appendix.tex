\begin{frame}{Backup: Pipeline Pseudocode}
\small
\textbf{Experiment loop skeleton}
\begin{enumerate}
  \item Generate \texttt{text}, \texttt{pattern} for a scenario.
  \item Run all algorithms and collect \texttt{(comparisons, match, wall\_sec)}.
  \item Append row records with scenario metadata.
  \item Persist section CSV, then aggregate means by algorithm.
  \item Plot comparisons and wall-clock for cross-method interpretation.
\end{enumerate}
\end{frame}
